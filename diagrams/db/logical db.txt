@startuml
hide circle
skinparam nodesep 60
skinparam ranksep 80

' Описание сущностей
entity "Мастер" as Master {
    + Идентификатор : Числовой <<PK>>
    --
    Имя : Текстовый
    Специализация : Текстовый
    Опыт работы (лет) : Числовой
    Телефон : Текстовый
    Фотография : Файловый
    Работает в салоне : Логический
}


entity "Пользователь системы" as User {
    + Идентификатор : Числовой <<PK>>
    --
    Логин : Текстовый
    Пароль : Текстовый
    Права администратора : Логический
}

entity "Услуга" as Service {
    + Идентификатор : Числовой <<PK>>
    --
    Название услуги : Текстовый
    Описание : Текстовый
    Стоимость : Вещественный
    Длительность : Временной
}

entity "Клиент" as Client {
    + Идентификатор : Числовой <<PK>>
    --
    Имя : Текстовый
    Дата рождения : Дата
    Телефон : Текстовый
    Электронная почта : Текстовый
    Дата регистрации : Дата и время
    Согласие родителей : Файловый
}

entity "Запись" as Appointment {
    + Идентификатор : Числовой <<PK>>
    --
    Дата и время приема : Дата и время
    Статус записи : Текстовый
    Дата создания записи : Дата и время
    Эскиз (изображение) : Файловый
    Примечание : Текстовый
}

entity "Работа мастера" as MasterWork {
    + Идентификатор : Числовой <<PK>>
    --
    Название работы : Текстовый
    Описание работы : Текстовый
    Изображение : Файловый
    Дата добавления : Дата и время
    Категория : Текстовый
    Часть тела : Текстовый
}

entity "Отпуск" as Vacation {
    + Идентификатор : Числовой <<PK>>
    --
    Дата начала : Дата
    Дата окончания : Дата
    Причина : Текстовый
}

entity "Расписание работы" as Availability {
    + Идентификатор : Числовой <<PK>>
    --
    День недели : Числовой
    Время начала работы : Временной
    Время окончания работы : Временной
    Активность : Логический
}

' РАСПРЕДЕЛЕНИЕ СВЯЗЕЙ (управление геометрией)

' Пользователь над Мастером
User -down-|| Master

' Служебные таблицы Мастера (уводим вправо и вверх)
Master ||-right-o{ MasterWork
Master ||-up-o{ Vacation
Master ||-up-o{ Availability

' Основная логика (внизу)
Master ||--o{ Appointment
Service ||-left-o{ Appointment
Client ||-right-o{ Appointment

' Связь многие-ко-многим (сбоку)
Master }o-left-o{ Service
@enduml
